#ATTACKMODE SERIAL ECM_ETHERNET # Uncomment for debugging purposes. Faster boot without.

# Adding CPU Throttling to prevent overheating.
CUCUMBER ENABLE

targetPayload = 0
i = 0

GET SWITCH_POSITION

while [[ $SWITCH_POSITION = switch2 ]]; do
	for i in R G B Y C M W OFF; do
		if [[ -d /root/udisk/payloads/payload_${i}_empty ]]; then
			continue
		fi
		LED $i
		sleep 1
		GET SWITCH_POSITION
		if [[ $SWITCH_POSITION = switch1 ]]; then
			break
		fi
	done
done

# Starting Blinking for Copy and Solid for DONE
LED $i 100
rm -rf /root/udisk/payloads/switch1/*
cp -R /root/udisk/payloads/payload_${i}_*/* /root/udisk/payloads/switch1/

# Improved LED Indication on new payload. Adds to 2nd/3rd Line of Payload. 
sed -i '2G' /root/udisk/payloads/switch1/payload.txt
sed -i "2i# Adding LED Indication to Hop-Swapped Payload." /root/udisk/payloads/switch1/payload.txt
sed -i "3iLED $i 100" /root/udisk/payloads/switch1/payload.txt
sed -i "4isleep 2" /root/udisk/payloads/switch1/payload.txt
sleep .5

# Added LED Indication for Syncing the Disk Partition.
sync
LED $i VERYFAST
wait
LED $i SOLID

# Adding Shutdown script with bonus "Wait 3 sec, blink twice and shutdown".
sleep 3
LED OFF
LED $i SOLID
LED OFF
LED $i SOLID
LED OFF
LED $i SOLID
shutdown 0
